## Climbing Stairs 
```
class Solution {
    public int climbStairs(int n) {
        int[] dp = new int[n + 1];
        dp[0] = 1;
        dp[1] = 1;
        for(int i = 2 ; i <= n ; i++){
            dp[i] = dp[i - 1] + dp[ i - 2];
        }
        return dp[n];
    }
}
```
## Frog Jump
Input: heights = [2, 1, 3, 5, 4]
Output: 2
Explanation:
One possible route can be,
0th step -> 2nd Step = abs(2 - 3) = 1
2nd step -> 4th step = abs(3 - 4) = 1
Total = 1 + 1 = 2.
```
class Solution {
    public int frogJump(int[] heights) {
        int n = heights.length;
        int prev = 0;
        int prev2 = 0;
        for(int i = 1 ; i < n ; i++){
            int j1 = prev + Math.abs(heights[i] - heights[i - 1]);
            int j2 = Integer.MAX_VALUE;
            if(i > 1){
                j2 = prev2 + Math.abs(heights[i] - heights[ i - 2]);
            }
            int curr = Math.min(j1 , j2);
            prev2 = prev;
            prev = curr;
        }
        return prev;
    }
}
```
## Frog Jumps K distance 
Input: heights = [10, 5, 20, 0, 15], k = 2
Output: 15
Explanation:
0th step -> 2nd step, cost = abs(10 - 20) = 10
2nd step -> 4th step, cost = abs(20 - 15) = 5
Total cost = 10 + 5 = 15.
```
class Solution {
    public int frogJump(int[] heights, int k) {
        int n = heights.length;
        int[] dp = new int[n + 1];
        Arrays.fill(dp , -1);
        dp[0] = 0;
        for(int i = 1 ; i < n ; i++){
            int mmstep = Integer.MAX_VALUE;
            for(int j = 1; j <= k ; j++){
                if(i - j >= 0){
                    int jump = dp[i - j] + Math.abs(heights[i]-heights[i-j]);
                mmstep = Math.min(mmstep , jump);
                }
            }
            dp[i] = mmstep;
        }
        return dp[n - 1];
    }
}
```

## MAximum Sum of Non Adjacent Element

Input: nums = [1, 2, 4]
Output: 5
Explanation:
[1, 2, 4], the underlined elements are taken to get the maximum sum.

```
class Solution {
    public int func(int[] nums , int n , int[] dp){
        if(n == 0){
            return nums[n];
        }
        if(n < 0){
            return 0;
        }
        if(dp[n] != -1){
            return dp[n];
        }
        int pick = nums[n] + func(nums,n-2,dp);
        int nonpick = func(nums,n-1,dp);
        dp[n] = Math.max(pick , nonpick);
        return dp[n];
    }
    public int nonAdjacent(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n + 1];
        Arrays.fill(dp , -1);
        return func(nums , n - 1, dp);
    }
}
```

class Solution {
    public int houseRobber(int[] money) {
        int n = money.length;
        if(n == 0){
            return 0;
        }
        if(n == 1){
            return money[0];
        }
        int[] arr1 = new int[n - 1];
        int[] arr2 = new int[n - 1];
        for(int i = 0 ; i < n ; i++){
            if(i != n - 1){
                arr1[i] = money[i];
            }
            if(i != 0){
                arr2[i - 1] = money[i];
            }
        }
        int dp1[] = new int[n - 1];
        int dp2[] = new int[n - 1];
        Arrays.fill(dp1 , -1);
        Arrays.fill(dp2 , -1);
        int ans1 = func(arr1 , n - 2 , dp1);
        int ans2 = func(arr2 , n - 2 , dp2);
        return Math.max(ans1 , ans2);
    }
    public int func(int[] arr , int n , int[] dp){
        if(n < 0){
            return 0;
        }
        if(n == 0){
            return arr[0];
        }
        if(dp[n] != -1){
            return dp[n];
        }
        int pick = arr[n] + func(arr,n-2,dp);
        int nonpick = func(arr,n-1,dp);
        dp[n] = Math.max(pick , nonpick);
        return dp[n];
    }
}
```

## Minimum Cost Jumps
Input: cost = [10, 15, 20]
Output: 15
Explanation: Start at index 1, pay 15, jump 2 steps → done.
Input: cost = [1, 100, 1, 1, 1, 100, 1, 1, 100, 1]
Output: 6
```
import java.util.*;
class A{
    public static int solve(int[] arr , int n , int[] dp){
        if(n == 0){
            return arr[0];
        }
        if(n == 1){
            return arr[1];
        }
        if(dp[n] != -1){
            return dp[n];
        }
        dp[n] = arr[n] + Math.min(solve(arr,n-1,dp),solve(arr,n-2,dp));
        return dp[n];
    }
    
    public static void main(String args[]){
        int arr[] = {10,15,20};
        int n = arr.length;
        int[] dp = new int[n + 1];
        Arrays.fill(dp , -1);
        System.out.println(Math.min(solve(arr,n-1,dp) , solve(arr,n-2,dp)));
    }
}
